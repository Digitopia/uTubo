00234 int receiveAnalogRead()
00235 {
00236                 unsigned char low, high;
00237 #if defined(ADCSRA) && defined(ADCL)
00238                 // ADSC is cleared when the conversion finishes
00239                 while (bit_is_set(ADCSRA, ADSC))
00240                                 ;
00241 
00242                 // we have to read ADCL first; doing so locks both ADCL
00243                 // and ADCH until ADCH is read.  reading ADCL second would
00244                 // cause the results of each conversion to be discarded,
00245                 // as ADCL and ADCH would be locked when it completed.
00246                 low  = ADCL;
00247                 high = ADCH;
00248 #else
00249                 // we dont have an ADC, return 0
00250                 low  = 0;
00251                 high = 0;
00252 #endif
00253 
00254                 // combine the two bytes
00255                 return (high << 8) | low;
